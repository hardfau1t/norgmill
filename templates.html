Norg HTML Templates

This document contains HTML templates for all possible text representations in the Norg format as described in the specification.

Base (Document) Template

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>{{title}}</title>
    <link rel="stylesheet" href="/assets/styles.css">
</head>
<body>
    <div class="norg-content">
        {{{body}}}
    </div>
</body>
</html>

Document Metadata

<div class="document-meta">
    {{#if title}}<h1 class="document-title">{{title}}</h1>{{/if}}
    {{#if author}}<div class="document-author">By: {{author}}</div>{{/if}}
    {{#if date}}<div class="document-date">{{date}}</div>{{/if}}
    {{#if description}}<div class="document-description">{{description}}</div>{{/if}}
</div>

Detached Modifiers

Structural Detached Modifiers

Heading

<div class="heading-block level-{{level}}">
    <h{{level}} id="heading-{{id}}">{{{title}}}</h{{level}}>
    {{#if content}}
    <div class="heading-content">
        {{{content}}}
    </div>
    {{/if}}
</div>

Nestable Detached Modifiers

Unordered List

<ul class="norg-unordered-list level-{{level}}">
    {{#each items}}
    <li>
        <div class="list-item-content">{{{content}}}</div>
        {{#if nested}}
        {{{nested}}}
        {{/if}}
    </li>
    {{/each}}
</ul>

Ordered List

<ol class="norg-ordered-list level-{{level}}">
    {{#each items}}
    <li>
        <div class="list-item-content">{{{content}}}</div>
        {{#if nested}}
        {{{nested}}}
        {{/if}}
    </li>
    {{/each}}
</ol>

Quote

<div class="quote-block level-{{level}}">
    <blockquote>
        <div class="quote-content">{{{content}}}</div>
        {{#if nested}}
        <div class="nested-quotes">
            {{{nested}}}
        </div>
        {{/if}}
    </blockquote>
</div>

Range-able Detached Modifiers

Definition

<dl class="norg-definition">
    <dt>{{{term}}}</dt>
    <dd>{{{definition}}}</dd>
</dl>

Footnote

<div class="footnote" id="footnote-{{id}}">
    <div class="footnote-label">{{{title}}}</div>
    <div class="footnote-content">{{{content}}}</div>
</div>

Footnote Reference

<a href="#footnote-{{id}}" class="footnote-ref"><sup>{{{label}}}</sup></a>

Table

<table class="norg-table">
    {{#if caption}}
    <caption>{{{caption}}}</caption>
    {{/if}}
    <tbody>
        {{#each rows}}
        <tr>
            {{#each this}}
            <td>{{{this}}}</td>
            {{/each}}
        </tr>
        {{/each}}
    </tbody>
</table>
